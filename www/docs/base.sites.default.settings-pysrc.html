 42  # The configuration file to be loaded is based upon the rules below. 
 43  # 
 44  # The configuration directory will be discovered by stripping the 
 45  # website's hostname from left to right and pathname from right to 
 46  # left. The first configuration file found will be used and any 
 47  # others will be ignored. If no other configuration file is found 
 48  # then the default configuration file at 'sites/default' will be used. 
 49  # 
 50  # For example, for a fictitious site installed at 
 51  # http://www.drupal.org/mysite/test/, the 'settings.php' 
 52  # is searched in the following directories: 
 53  # 
 54  #  1. sites/www.drupal.org.mysite.test 
 55  #  2. sites/drupal.org.mysite.test 
 56  #  3. sites/org.mysite.test 
 57  # 
 58  #  4. sites/www.drupal.org.mysite 
 59  #  5. sites/drupal.org.mysite 
 60  #  6. sites/org.mysite 
 61  # 
 62  #  7. sites/www.drupal.org 
 63  #  8. sites/drupal.org 
 64  #  9. sites/org 
 65  # 
 66  # 10. sites/default 
 67  # 
 68  # If you are installing on a non-standard port number, prefix the 
 69  # hostname with that number. For example, 
 70  # http://www.drupal.org:8080/mysite/test/ could be loaded from 
 71  # sites/8080.www.drupal.org.mysite.test/.

 75  # Database settings: 
 76  # 
 77  # Note that the $db_url variable gets parsed using PHP's built-in 
 78  # URL parser (i.e. using the "parse_url()" function) so make sure 
 79  # not to confuse the parser. If your username, password 
 80  # or database name contain characters used to delineate 
 81  # $db_url parts, you can escape them via URI hex encodings: 
 82  # 
 83  #   : = %3a   / = %2f   @ = %40 
 84  #   + = %2b   ( = %28   ) = %29 
 85  #   ? = %3f   = = %3d   & = %26 
 86  # 
 87  # To specify multiple connections to be used in your site (i.e. for 
 88  # complex custom modules) you can also specify an associative array 
 89  # of $db_url variables with the 'default' element used until otherwise 
 90  # requested. 
 91  # 
 92  # You can optionally set prefixes for some or all database table names 
 93  # by using the $db_prefix setting. If a prefix is specified, the table 
 94  # name will be prepended with its value. Be sure to use valid database 
 95  # characters only, usually alphanumeric and underscore. If no prefixes 
 96  # are desired, leave it as an empty string ''. 
 97  # 
 98  # To have all database names prefixed, set $db_prefix as a string: 
 99  # 
100  #   $db_prefix = 'main_'; 
101  # 
102  # To provide prefixes for specific tables, set $db_prefix as an array. 
103  # The array's keys are the table names and the values are the prefixes. 
104  # The 'default' element holds the prefix for any tables not specified 
105  # elsewhere in the array. Example: 
106  # 
107  #   $db_prefix = array( 
108  #     'default'   => 'main_', 
109  #     'users'     => 'shared_', 
110  #     'sessions'  => 'shared_', 
111  #     'role'      => 'shared_', 
112  #     'authmap'   => 'shared_', 
113  #     'sequences' => 'shared_', 
114  #   ); 
115  # 
116  # Database URL format: 
117  #   $db_url = 'mysql://username:password@localhost/databasename'; 
118  #   $db_url = 'mysqli://username:password@localhost/databasename'; 
119  #   $db_url = 'pgsql://username:password@localhost/databasename'; 
120  # 
121  db_url = 'mysqli://main@localhost/drupalhead'; 
122  db_prefix = '';

209  # reverse_proxy accepts a boolean value. 
210  # 
211  # Enable this setting to determine the correct IP address of the remote 
212  # client by examining information stored in the X-Forwarded-For headers. 
213  # X-Forwarded-For headers are a standard mechanism for identifying client 
214  # systems connecting through a reverse proxy server, such as Squid or 
215  # Pound. Reverse proxy servers are often used to enhance the performance 
216  # of heavily visited sites and may also provide other site caching, 
217  # security or encryption benefits. If this Drupal installation operates 
218  # behind a reverse proxy, this setting should be enabled so that correct 
219  # IP address information is captured in Drupal's session management, 
220  # logging, statistics and access management systems; if you are unsure 
221  # about this setting, do not have a reverse proxy, or Drupal operates in 
222  # a shared hosting environment, this setting should be set to disabled. 
223  # 
224  #   'reverse_proxy' => TRUE, 
225  # 
226  # reverse_proxy accepts an array of IP addresses. 
227  # 
228  # Each element of this array is the IP address of any of your reverse 
229  # proxies. Filling this array Drupal will trust the information stored 
230  # in the X-Forwarded-For headers only if Remote IP address is one of 
231  # these, that is the request reaches the web server from one of your 
232  # reverse proxies. Otherwise, the client could directly connect to 
233  # your web server spoofing the X-Forwarded-For headers. 
234  # 
235  #   'reverse_proxy_addresses' => array('a.b.c.d', ...), 
236  # );
