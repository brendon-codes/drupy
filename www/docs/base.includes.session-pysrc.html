 47    # So we are moving session closure before destructing objects. 
 48    register_shutdown_function('session_write_close') 
 49    # Handle the case of first time visitors and clients that don't store cookies (eg. web crawlers). 
 50    if (not p.isset(_COOKIE, p.session_name())): 
 51      user = drupal_anonymous_user() 
 52      return '' 
 53    # Otherwise, if the session is still active, we have a record of the client's session in the database. 
 54    user = db_fetch_object(db_query("SELECT u.*, s.* FROM {users} u INNER JOIN {sessions} s ON u.uid = s.uid WHERE s.sid = '%s'", key)) 
 55    # We found the client's session record and they are an authenticated user 
 56    if (user and user.uid > 0): 
 57      # This is done to unserialize the data member of user 
 58      user = drupal_unpack(user) 
 59      # Add roles element to user 
 60      user.roles = array() 
 61      user.roles[DRUPAL_AUTHENTICATED_RID] = 'authenticated user' 
 62      result = db_query("SELECT r.rid, r.name FROM {role} r INNER JOIN {users_roles} ur ON ur.rid = r.rid WHERE ur.uid = %d", user.uid) 
 63      while True: 
 64        role = db_fetch_object(result) 
 65        if role == None: 
 66          break 
 67        user.roles[role.rid] = role.name 
 68    # We didn't find the client's record (session has expired), or they are an anonymous user. 
 69    else: 
 70      session = (user.session if p.isset(user.session) else '') 
 71      user = drupal_anonymous_user(session)
